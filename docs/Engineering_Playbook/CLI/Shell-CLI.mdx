## Make the script executable

```shell
xed hello.sh
echo Hello $1
chmod +x hello.sh
./hello.sh
./hello.sh Rebecca
./hello.sh Rebecca Staveley-Taylor
```

## Script 1 - Prompts & Variables

```shell
#!/bin/bash
firstname=$1
surname=$2

echo -n "Hello, what is your name?"
read firstname
echo -n "Thank you, and what is your surname?"
reach surname
clear
if ["$firstname" == "Rebecca"] && ["$surname" == "Staveley-Taylor"]
then echo "Great name, " $firstname $surname
else echo Hello $firstname $surname, how are you on this fine $(date +%A)?
fi
```

## Script 2 - Conditionals
```shell
#!/bin/bash

count =0

while [$count -lt 100];do
  echo $count
  let count=count+1
  done
  
for count in {0...100};do
  echo $count
  let count=count+1case
  done
    
until [$count -gt 100];do
  echo $count
  let count=count+1
  done

for count in {0...10};do
  mkdir Foolder$count
  let count=count+1case
  done
  
for count in {0...100...5};do
  echo $count
  let count=count+1case
  done
```

## Script 3 - Choices & Loops

```shell
#!/bin/bash

choice=4

echo "1. Lemon"
echo "2. is"
echo "3. sour"
echo -n "PLease choose an option (1, 2, or 3)"

while [$choice -eq 4]; do
  
  read choice
  
  if [$choice -eq 1]; then
    echo "You have chosen: Lemon"
  else
     if [$choice -eq 2]; then
       echo "You have chosen: Is"
     else
       if [$choice -eq 3]; then
         echo "You have chosen: Awesome"
       else
         echo "Please make a choice between 1 to 3"
         echo "1. Lemon"
         echo "2. is"
         echo "3. sour"
         echo -n "PLease choose an option (1, 2, or 3)"
         choice=4
       fi 
     fi 
  fi 
  done
```
